c0z (Impetus Omnia Omnem Impetum) March 8, 2023, 1:10pm 3

You forgot to execute the command in the variable $STR. If you do:

# `$STR`
ssh 159.202.233.107
#

#!/bin/bash
RANDOM=25
for i in {1..100}
do
        STR="printf %d.%d.%d.%d\n $((RANDOM % 256)) $((RANDOM % 256)) $((RANDOM % 256)) $((RANDOM % 256))"
done
while true; do
    read -p "Start SSH connection? " yn
    case $yn in
        [Yy]* )
                echo ssh `$STR`
                ;;
        [Nn]* )
                exit
                ;;
        * )
                echo "Invalid Input"
                ;;
    esac
done
